        class Solution {

    int maxAliveYear(vector<int>& birth, vector<int>& death) {
        vector<int> res(110, 0);
        int n = birth.size();
        //ç”Ÿäº1908å¹´ã€æ­»äº1909å¹´çš„äººåº”å½“è¢«åˆ—å…¥1908å¹´å’Œ1909å¹´çš„è®¡æ•°

        }
        int ans = 0;
        //å¹´æ•°çš„è¾ƒå°å€¼

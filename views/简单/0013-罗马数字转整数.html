            class Solution{
    public int romanToInt(String s){
        int res=0;
        if (s==null || s.length()==0)
            return res;
        int len=s.length()-1;
        for (int i=len;i>=0;i--){
            char c=s.charAt(i);
            switch (c){
                case 'M':
                    res+=1000;
                    break;
                case 'D':
                    res+=500;
                    break;
                case 'C':
                    res+=100*(res>=500?-1:1);
                    break;
                case 'L':
                    res+=50;
                    break;
                case 'X':
                    res+=10*(res>=50?-1:1);
                    break;
                case 'V':
                    res+=5;
                    break;
                case 'I':
                    res+=(res>=5?-1:1);
                    break;
                    default:break;
            }
        }
        return res;
    }
}

            class Solution {
public:
    int romanToInt(string s) {
        unordered_map<string, int> m = {{"I", 1}, {"IV", 3}, {"IX", 8}, {"V", 5}, {"X", 10}, {"XL", 30}, {"XC", 80}, {"L", 50}, {"C", 100}, {"CD", 300}, {"CM", 800}, {"D", 500}, {"M", 1000}};
        int r = m[s.substr(0, 1)];
        for(int i=1; i<s.size(); ++i){
            string two = s.substr(i-1, 2);
            string one = s.substr(i, 1);
            r += m[two] ? m[two] : m[one];
        }
        return r;
    }
};

            # 中文题目

            从扑克牌中随机抽5张牌，判断是不是一个顺子，即这5张牌是不是连续的。2～10为数字本身，A为1，J为11，Q为12，K为13，而大、小王为 0 ，可以看成任意数字。A 不能视为 14。

             

            示例 1:

            输入: [1,2,3,4,5]
输出: True

             

            示例 2:

            输入: [0,0,1,2,5]
输出: True

             

            限制：

            数组长度为 5 

            数组的数取值为 [0, 13] .

            # 通过代码

            class Solution {
    public boolean isStraight(int[] nums) {
        Set<Integer> set = new HashSet<>();
        int max = 0;
        int min = 14;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) continue;
            // 存在重复牌，肯定不能构成顺子
            if (set.contains(nums[i])) return false;
            // 找牌起始位置
            min = Math.min(min, nums[i]);
            // 找牌最大位置
            max = Math.max(max, nums[i]);
            // 将牌加入set集合
            set.add(nums[i]);
        }
        // 如果牌之间相减小于5，说明肯定可以构成
        return max - min < 5;
    }
}

            class Solution {
    public boolean isStraight(int[] nums) {
        int totalZero = 0;
        // 设置最大的统计数组为16，因为我们是以最小的牌作为起点计算，那么大小王加上最大到15
        int[] arr = new int[16];
        // 设置最小的开始值为16
        int start = 16;
        // 统计，并找到最小值
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                totalZero++;
            } else {
                arr[nums[i]] = 1;
                start = Math.min(start, nums[i]);
            }
        }
        // 一共五张牌
        for (int i = 0; i < 5; i++) {
            // 超过了最大的区间，其实可以通过判断是否有重复跳过这个判断
            if (start > 15) {
                return false;
            }
            if (arr[start] == 1) {
                start++;
            } else if (arr[start] != 1 && totalZero > 0) {
                arr[start] = 1;
                start++;
                totalZero--;
            } else {
                return false;
            }
        }
        return true;
    }
}

            # 高赞题解

            # 解题思路：

            根据题意，此 555 张牌是顺子的 充分条件 如下：

            1. 除大小王外，所有牌 无重复 ；
            2. 设此 555 张牌中最大的牌为 maxmaxmax ，最小的牌为 minminmin （大小王除外），则需满足：

            max−min<5max - min < 5 max−min<5

            因而，可将问题转化为：此 555 张牌是否满足以上两个条件？

            {:width=600}

            # 方法一： 集合 Set + 遍历

            • 遍历五张牌，遇到大小王（即 000 ）直接跳过。
            • 判别重复： 利用 Set 实现遍历判重， Set 的查找方法的时间复杂度为 O(1)O(1)O(1) ；
            • 获取最大 / 最小的牌： 借助辅助变量 mamama 和 mimimi ，遍历统计即可。
            # 复杂度分析：
            • 时间复杂度 O(N)=O(5)=O(1)O(N) = O(5) = O(1)O(N)=O(5)=O(1) ： 其中 NNN 为 numsnumsnums 长度，本题中 N≡5N \equiv 5N≡5 ；遍历数组使用 O(N)O(N)O(N) 时间。
            • 空间复杂度 O(N)=O(5)=O(1)O(N) = O(5) = O(1)O(N)=O(5)=O(1) ： 用于判重的辅助 Set 使用 O(N)O(N)O(N) 额外空间。

            # 代码：
            class Solution:
    def isStraight(self, nums: List[int]) -> bool:
        repeat = set()
        ma, mi = 0, 14
        for num in nums:
            if num == 0: continue # 跳过大小王
            ma = max(ma, num) # 最大牌
            mi = min(mi, num) # 最小牌
            if num in repeat: return False # 若有重复，提前返回 false
            repeat.add(num) # 添加牌至 Set
        return ma - mi < 5 # 最大牌 - 最小牌 < 5 则可构成顺子

            class Solution {
    public boolean isStraight(int[] nums) {

        int max = 0, min = 14;
        for(int num : nums) {
            if(num == 0) continue; // 跳过大小王

        }
        return max - min < 5; // 最大牌 - 最小牌 < 5 则可构成顺子
    }
}

            # 方法二：排序 + 遍历

            • 先对数组执行排序。
            • 判别重复： 排序数组中的相同元素位置相邻，因此可通过遍历数组，判断 nums[i]=nums[i+1]nums[i] = nums[i + 1]nums[i]=nums[i+1] 是否成立来判重。
            • 获取最大 / 最小的牌： 排序后，数组末位元素 nums[4]nums[4]nums[4] 为最大牌；元素 nums[joker]nums[joker]nums[joker] 为最小牌，其中 jokerjokerjoker 为大小王的数量。
            # 复杂度分析：
            • 时间复杂度 O(Nlog⁡N)=O(5log⁡5)=O(1)O(N \log N) = O(5 \log 5) = O(1)O(NlogN)=O(5log5)=O(1) ： 其中 NNN 为 numsnumsnums 长度，本题中 N≡5N \equiv 5N≡5 ；数组排序使用 O(Nlog⁡N)O(N \log N)O(NlogN) 时间。
            • 空间复杂度 O(1)O(1)O(1) ： 变量 jokerjokerjoker 使用 O(1)O(1)O(1) 大小的额外空间。

            class Solution:
    def isStraight(self, nums: List[int]) -> bool:
        joker = 0
        nums.sort() # 数组排序
        for i in range(4):
            if nums[i] == 0: joker += 1 # 统计大小王数量
            elif nums[i] == nums[i + 1]: return False # 若有重复，提前返回 false
        return nums[4] - nums[joker] < 5 # 最大牌 - 最小牌 < 5 则可构成顺子

            class Solution {
    public boolean isStraight(int[] nums) {
        int joker = 0;
        Arrays.sort(nums); // 数组排序
        for(int i = 0; i < 4; i++) {
            if(nums[i] == 0) joker++; // 统计大小王数量
            else if(nums[i] == nums[i + 1]) return false; // 若有重复，提前返回 false
        }
        return nums[4] - nums[joker] < 5; // 最大牌 - 最小牌 < 5 则可构成顺子
    }
}

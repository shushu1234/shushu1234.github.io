package com.liuyao.sort;

import com.liuyao.utils.SortHelper;

/**
 * Created By liuyao on 2018/4/13 22:09.
 */
public class QuickSort &#123;
    private QuickSort()&#123;&#125;

    private static int partition(Comparable[] arr,int l,int r)&#123;
       Comparable v=arr[l];// 哨兵元素
       int j=l; //j初始为l，表示比哨兵小的元素个数为0
        for (int i = l+1; i <= r; i++) &#123;
            if (arr[i].compareTo(v)<0)&#123;
                j++; //找到比哨兵小的所有元素的最后的一个位置。
                swap(arr,j,i);  //交换，将比哨兵大的元素放到后面来
            &#125;
        &#125;
        swap(arr,l,j); //将哨兵和比它小的最后一个元素交换位置。
        return j;
    &#125;

    private static void sort(Comparable[] arr,int l,int r)&#123;
        if (l>=r)&#123;
            return;
        &#125;
        int p=partition(arr,l,r);
        sort(arr,1,p-1);
        sort(arr,p+1,r);
    &#125;

    public static void sort(Comparable[] arr)&#123;
        int n=arr.length;
        sort(arr,0,n-1);
    &#125;

    public static void swap(Object[] arr,int i,int j)&#123;
        Object temp=arr[i];
        arr[i]=arr[j];
        arr[j]=temp;
    &#125;

    public static void main(String[] args) &#123;
        Integer[] a=&#123;2,3,6,8,1,4,5,7&#125;;
        new QuickSort().sort(a);

    private QuickSort3() &#123;

    private static int partition(Comparable[] arr, int l, int r) &#123;
        swap(arr, l, (int) (Math.random() * (r - l + 1)) + l);

        Comparable v = arr[l];

        int i = l + 1, j = r;

        while (true) &#123;
//            当从前向后遇到大于等于v的数就停止
            while (i <= r && arr[i].compareTo(v) < 0) &#123;
                i++;
            &#125;

//            当从后向前遇到小于等于v的数就停止

            while (j >= l + 1 && arr[j].compareTo(v) > 0) &#123;
                j--;
            &#125;

            if (i > j) &#123;
                break;
            &#125;

            swap(arr, i, j);
            i++;
            j--;
        &#125;

        swap(arr, l, j);

    public static void sort(Comparable[] arr, int l, int r) &#123;
        if (r - l <= 15) &#123;
            new Insertion().sort(arr, l, r);
            return;
        &#125;
        swap(arr, l, (int) ((Math.random() * (r - l + 1)) + l));

        Comparable v = arr[l];
        int lt = l;   //arr[l+1...lt]<v
        int gt = r + 1; //arr[gt...r]>v
        int i = l + 1;  //arr[lt+1...i) == v

        while (i < gt) &#123;
            if (arr[i].compareTo(v) < 0) &#123;
                swap(arr, i, lt + 1);
                i++;
                lt++;
            &#125; else if (arr[i].compareTo(v) > 0) &#123;
                swap(arr, i, gt - 1);
                gt--;
            &#125; else &#123;
                i++;
            &#125;
        &#125;
        swap(arr, l, lt);
        sort(arr, l, lt - 1);
        sort(arr, gt, r);

    &#125;

6. 数组中的逆序对个数

主要是利用归并排序，比较，统计

package com.liuyao.sort;

import com.liuyao.utils.SortHelper;

import java.util.Arrays;

/**
 * Created By liuyao on 2018/4/15 21:18.
 */
public class InversionCount &#123;
    private InversionCount()&#123;&#125;

    private static long merge(Comparable[] arr,int l,int mid,int r)&#123;
        Comparable[] aux= Arrays.copyOfRange(arr,l,r+1);

//        初始化逆序对数个数为0L
        long res=0L;

//        初始化时，i指向左半边起始索引位置l，j指向右半边起始索引位置mid+1
        int i=l,j=mid+1;

        for (int k = l; k <=r ; k++) &#123;

//            左半边已经全部处理完
            if (i >mid)&#123;
                arr[k]=aux[j-l];
                j++;
            &#125; else if (j >r)&#123;  //右半边已经全部处理完
                arr[k]=aux[i-l];
                i++;
            &#125; else if (aux[i-l].compareTo(aux[j-l])<=0)&#123; //左边元素比右边小，将左边元素放入
                arr[k]=aux[i-l];
                i++;
            &#125;else &#123;   //否者右边的元素比左边的小，存在逆序对
                arr[k]=aux[j-l];
                j++;
                //此时这个元素和左半边部分的所有为处理的元素构成逆序对
                // 左半边部分此时为处理的元素个数为mid-j+l
                res+=(mid-i+l);
            &#125;
        &#125;
        return res;
    &#125;

    private static long solve(Comparable[] arr,int l,int r)&#123;
        if (l>=r)&#123;
            return 0L;
        &#125;
        int mid=l+(r-l)/2;
// 求出 arr[l...mid] 范围的逆序数
        long res1=solve(arr,l,mid);
// 求出 arr[mid+1...r] 范围的逆序数
        long res2=solve(arr,mid+1,r);

        return res1+res2+merge(arr,l,mid,r);
    &#125;

    public static long solve(Comparable[] arr)&#123;
        return solve(arr,0,arr.length-1);
    &#125;

    public static void main(String[] args) &#123;
        Integer[] a = &#123;2, 3, 6, 8, 1, 4, 5, 7&#125;;
        System.out.println(new InversionCount().solve(a));
    &#125;
